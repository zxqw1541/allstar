/* 대회 */
CREATE TABLE COMPETITION (
	cno int NOT NULL,  /* 대회번호 */
	eno int NOT NULL,  /* 종목번호 */
	ano int NOT NULL,  /* 연고지번호 */
	type varchar(50) NOT NULL,  /* 대회타입 */
	name varchar(50) NOT NULL,  /* 대회명 */
	team_num int NOT NULL,  /* 최대팀수 */
	join_num int NOT NULL DEFAULT 1,  /* 현참여팀수 */
	cost int NOT NULL DEFAULT 0,  /* 참가비 */
	sdt date NOT NULL,  /* 시작일 */
	edt date NOT NULL,  /* 종료일 */
	rsdt date NOT NULL,  /* 모집시작일 */
	redt datetime NOT NULL,  /* 모집종료일 */
	content text NOT NULL,  /* 내용 */
	poster varchar(255) NOT NULL,  /* 포스터 */
	prino int NOT NULL /* 상금분배번호 */
)
GO
